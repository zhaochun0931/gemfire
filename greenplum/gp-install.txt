install greenplum database in redhat linux


greenplum-db-6.24.3-rhel8-x86_64.rpm


# add group and user
groupadd gpadmin
useradd gpadmin -r -m -g gpadmin
passwd gpadmin


# intall greenplum database
yum install ./greenplum-db-6.24.3-rhel8-x86_64.rpm


# change permission
chown -R gpadmin:gpadmin /usr/local/greenplum*










# create the directory

mkdir -p /data/master /data/primary /data/mirror
chown gpadmin:gpadmin /data/master /data/primary /data/mirror
